% A useful and fancy collection of Latex packages
% Describes the function of each package briefly.

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Font packages: input, output, etc.
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\usepackage[T1]{fontenc}
\usepackage[utf8]{inputenc}

%Latin modern font
\usepackage{lmodern}

%embolden text
\usepackage{bm}

%colored fonts
\usepackage{color}

%Stretches and shrinks fonts to fit on line a little better
\usepackage{microtype}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Standard AMS and math packages
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%amsmath: miscellaneous enhancements for mathematical documents.
\usepackage{amsmath}

%amsthm: Package for theorem-like structures: theorem, lemma, proof, corollary, etc.
\usepackage{amsthm}

%amssymb: extension of symbol collection.
\usepackage{amssymb}

%mathtools: Extends and fixes deficiencies in amsmath
\usepackage[showonlyrefs]{mathtools}
%showonlyrefs: Only show equation numbers when cross referenced somewhere

%array and tabular environment
\usepackage{array}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Miscellaneous:
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%authblk: author/affiliation and footnote style authors.
\usepackage{authblk}

%Paper size and page margins
\usepackage{geometry}

%Fancy headers on every page: Section and subsection on left and right respectively
%Fancy headers on every page: Section and subsection on left and right respectively
\usepackage{fancyhdr, blindtext}
\newcommand{\changefont}{\fontsize{7}{9}\selectfont}
\fancyhf{}
\fancyhead[R]{\changefont \slshape \rightmark} %section
\fancyhead[L]{\changefont \slshape \textbf{\leftmark}} %chapter
\fancyfoot[C]{\changefont \thepage} %footer
\pagestyle{fancy}

%Fixed-length columns with L, C, R alignment.
\newcolumntype{L}[1]{>{\raggedright\let\newline\\\arraybackslash\hspace{0pt}}m{#1}}
\newcolumntype{C}[1]{>{\centering\let\newline\\\arraybackslash\hspace{0pt}}m{#1}}
\newcolumntype{R}[1]{>{\raggedleft\let\newline\\\arraybackslash\hspace{0pt}}m{#1}}

%Todonotes package: Useful for commenting in collaboration
\usepackage[colorinlistoftodos]{todonotes}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Referencing and bibliography
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%biblatex: Bibliography package
%biber: Recommended backend
%ieee: Nice looking style
%url = false: Suppresses URLs for non websites. Doesn't seem to work for arXiv.
\usepackage[backend=biber,style=ieee]{biblatex}


%Hyperref: set up cross reference and citation links
%Also creates bookmarks for sections, subsections, etc.
\usepackage{hyperref}
\hypersetup{colorlinks=true, linkcolor=blue, citecolor=blue, urlcolor=blue,bookmarksnumbered,bookmarksdepth=3}
